cmake_minimum_required(VERSION 3.2)
project(myServer)
# open62541 must be installed.
# If in custom path, then use -DCMAKE_PREFIX_PATH=/home/ubuntu/conneXt/install
# To force cmake to use python3, use -DPYTHON_EXECUTABLE:FILEPATH=/usr/bin/python3.8 
find_package(open62541 1.3.9 REQUIRED COMPONENTS FullNamespace)
# Output directory for Nodeset Compiler
set(GENERATE_OUTPUT_DIR "${CMAKE_BINARY_DIR}/src_generated/")
file(MAKE_DIRECTORY "${GENERATE_OUTPUT_DIR}")
include_directories("${GENERATE_OUTPUT_DIR}")
ua_generate_datatypes(
	NAME "ua_types_di"
	TARGET_SUFFIX "types-di"
	NAMESPACE_MAP "1:http://opcfoundation.org/UA/DI/"
	FILE_CSV "${PROJECT_SOURCE_DIR}/nodesets/DI/OpcUaDiModel.csv"
	FILES_BSD "${PROJECT_SOURCE_DIR}/nodesets/DI/Opc.Ua.Di.Types.bsd"
)
ua_generate_nodeset(
	NAME "di"
	FILE "${PROJECT_SOURCE_DIR}/nodesets/DI/Opc.Ua.Di.NodeSet2.xml"
	TYPES_ARRAY "UA_TYPES_DI"
	INTERNAL
	DEPENDS_TYPES "UA_TYPES"
	DEPENDS_NS "${PROJECT_SOURCE_DIR}/nodesets/Schema/Opc.Ua.NodeSet2.xml"
	DEPENDS_TARGET "open62541-generator-types-di"
)
ua_generate_datatypes(
	NAME "ua_types_rob"
	TARGET_SUFFIX "types-rob"
	NAMESPACE_MAP "2:http://opcfoundation.org/UA/Robotics/"
	FILE_CSV "${PROJECT_SOURCE_DIR}/nodesets/Robotics/NodeIds.csv"
	FILES_BSD "${PROJECT_SOURCE_DIR}/nodesets/Robotics/Opc.Ua.Robotics.Types.bsd"
)
ua_generate_nodeset(
	NAME "rob"
	FILE "${PROJECT_SOURCE_DIR}/nodesets/Robotics/Opc.Ua.Robotics.NodeSet2.xml"
	TYPES_ARRAY "UA_TYPES_ROB"
	INTERNAL
	DEPENDS_TYPES "UA_TYPES" "UA_TYPES_DI"
	DEPENDS_NS "${PROJECT_SOURCE_DIR}/nodesets/Schema/Opc.Ua.NodeSet2.xml" "${PROJECT_SOURCE_DIR}/nodesets/DI/Opc.Ua.Di.NodeSet2.xml"
	DEPENDS_TARGET "open62541-generator-types-rob"
)
ua_generate_datatypes(
	NAME "ua_types_ff"
	TARGET_SUFFIX "types-ff"
	NAMESPACE_MAP "3:https://nextusc.com/UA/FF/"
	FILE_CSV "${PROJECT_SOURCE_DIR}/model/OpcUaFfModel.csv"
	FILES_BSD "${PROJECT_SOURCE_DIR}/model/Published/Opc.Ua.Ff.Types.bsd"
)
ua_generate_nodeset(
	NAME "ff"
	FILE "${PROJECT_SOURCE_DIR}/model/Published/Opc.Ua.Ff.NodeSet2.xml"
	TYPES_ARRAY "UA_TYPES_FF"
	INTERNAL
	DEPENDS_TYPES "UA_TYPES" "UA_TYPES_DI" "UA_TYPES_ROB"
	DEPENDS_NS "${PROJECT_SOURCE_DIR}/nodesets/Schema/Opc.Ua.NodeSet2.xml" "${PROJECT_SOURCE_DIR}/nodesets/DI/Opc.Ua.Di.NodeSet2.xml" "${PROJECT_SOURCE_DIR}/nodesets/Robotics/Opc.Ua.Robotics.NodeSet2.xml"
	DEPENDS_TARGET "open62541-generator-ns-di" "open62541-generator-ns-rob" "open62541-generator-types-ff"
)
# Previous macro automatically sets some variables which hold the generated source code files using the provided NAME
add_executable(myServer
               ${UA_NODESET_DI_SOURCES}
               ${UA_TYPES_DI_SOURCES}
               ${UA_NODESET_ROB_SOURCES}
               ${UA_TYPES_ROB_SOURCES}
               ${UA_NODESET_FF_SOURCES}
               ${UA_TYPES_FF_SOURCES}
               myServer.c
               )
# Make sure the nodeset compiler is execute before compiling the main file

target_link_libraries(myServer open62541::open62541)
